def inputHandler():
    while True:
        try:
            user_input = input("~ ")
            # Allow user to exit the loop
            if user_input.lower().strip() in ["exit", "quit"]:
                print("Exiting...")
                break
            # Check for valid directions
            elif user_input.lower().strip() in ["up", "down", "left", "right"]:
                updatingPlayboard(user_input)
            else:
                raise ValueError
        except ValueError:
